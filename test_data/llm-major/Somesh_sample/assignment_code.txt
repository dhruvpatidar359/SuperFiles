#include<iostream>

// class MATRIX
class Matrix {
    // m is row and n is columns
    int m=3 , n=3;
    int a[3][3] ,  b[3][3] , c[3][3] , sum[3][3]={0};


    public:

    // Show Matrix
    void output(){
        for(int i=0 ; i<m ; i++){
            for(int j=0 ; j<n ; j++){
                cout<<sum[i][j]<<" ";
            }
            cout<<endl;
        }
    }


    // Input one Matrix
    void In_one(){
        cout<<"Enter First Matrix:"<<endl;
        for(int i=0 ; i<m ; i++){
            for(int j=0 ; j<n ; j++){
                cin>>a[i][j];
            }
        }
    }

    //Input two Matrix
    void In_two(){
        cout<<"Enter First Matrix:"<<endl;
        for(int i=0 ; i<m ; i++){
            for(int j=0 ; j<n ; j++){
                cin>>a[i][j];
            }
        }
        cout<<"Enter Second Matrix:"<<endl;
        for(int i=0 ; i<m ; i++){
            for(int j=0 ; j<n ; j++){
                cin>>b[i][j];
            }
        }
    }
    


    //Addition
    void addition(){
        In_two();
        for(int i=0 ; i<m ; i++){
            for(int j=0 ; j<n ; j++){
                sum[i][j]=a[i][j]+b[i][j];
            }
        }
        output();
    }

    //Substraction
    void substraction() {
        In_two();
        for(int i=0 ; i<m ; i++){
            for(int j=0 ; j<n ; j++){
                sum[i][j]=a[i][j]-b[i][j];
            }
        }
        output();
    }

    //Multiplication
    void multiplication() {
        In_two();
        for(int i=0 ; i<m ; i++){
            for(int j=0 ; j<n ; j++){
                for(int k=0 ; k<n ; k++){
                    sum[i][j]=sum[i][j]+a[i][k]*b[k][j];
                }
            }
        }
        output();
    }


    //Determinant
    int determinant(){
        In_one();
        int sum=0;
        int p=co_factor(0,0);
        int q=co_factor(0,1);
        int r=co_factor(0,2);
        sum=a[0][0]*p - a[0][1]*q + a[0][2]*r;
        cout<<"Determinant is "<<sum<<endl;
        return sum;


    }


    // minor co-factor
    int co_factor(int r , int c){
        int k=0;
        int co_fac[4];
        for(int i=0 ; i<m ; i++){
            for(int j=0 ; j<n ; j++){
                if(i!=r and j!=c){
                    co_fac[k]=a[i][j];
                    ++k;
                }
            }
        }
        return ((co_fac[0]*co_fac[3]) - (co_fac[1]*co_fac[2]));
    }

    // Inverse
    void Inverse(){
        int deter=determinant();
        if(deter==0) cout<<"Inverse Does Not Exist";
        else{
            float inverse[m][n];
            for(int i=0 ; i<m ; i++){
                for(int j=0 ; j<n ; j++){
                    int co_fac;
                    if((i+j)%2!=0){
                        co_fac=co_factor(i,j)*-1;
                    }
                    else co_fac=co_factor(i,j);
                    inverse[j][i]=co_fac;
                }
            }
            for(int i=0 ; i<m ; i++){
                for(int j=0 ; j<n ; j++){
                    cout<<inverse[i][j]/deter<<" ";
                }
                cout<<endl;
            }
        }
    }

    // Transpose
    void Transpose(){
        In_one();
        int transpose[m][n];
        for(int i=0 ; i<m ; i++){
            for(int j=0 ; j<n ; j++){
                transpose[j][i]=a[i][j];
            }
        }
        for(int i=0 ; i<m ; i++){
            for(int j=0 ; j<n ; j++){
                cout<<transpose[i][j]<<" ";
            }
            cout<<endl;
        }
    }


    // TRACE
    void Trace(){
        In_one();

                }
            }
        }
